macro(pythonize_bool truth_var var)
  if(${truth_var})
    set(${var} True)
  else()
    set(${var} False)
  endif()
endmacro()

function(configure_cucorr_lit_site_cfg input output)
  set(LIT_SITE_CFG_IN_HEADER
      "## Autogenerated for cucorr from ${input}\n## Do not edit!"
  )
  
  set(CUCORR_PROJECT_DIR ${PROJECT_SOURCE_DIR})
  set(CUCORR_LIBRARY_DIR ${CMAKE_BINARY_DIR}/lib)
  set(CUCORR_PASS_DIR ${CUCORR_LIBRARY_DIR}/pass)
  set(CUCORR_RUNTIME_DIR ${CUCORR_LIBRARY_DIR}/runtime)
  set(CUCORR_SCRIPT_DIR ${CMAKE_BINARY_DIR}/scripts)

  if(${LLVM_VERSION_MAJOR} VERSION_GREATER_EQUAL "13")
    set(CUCORR_OPT_ARGS "-enable-new-pm=0")
  endif()

  cucorr_target_generate_file(${input} ${output})
endfunction()

function(cucorr_add_lit_testsuite target comment)
  cmake_parse_arguments(ARG "" "" "PARAMS;DEPENDS;ARGS" ${ARGN})

  foreach(param ${ARG_PARAMS})
    list(APPEND TEST_PARAMS --param ${param})
  endforeach()

  add_custom_target(${target}
    COMMAND ${CUCORR_LIT_EXEC} ${ARG_ARGS} ${TEST_PARAMS} ${ARG_UNPARSED_ARGUMENTS}
    COMMENT "${comment}"
    USES_TERMINAL
  )

  if (ARG_DEPENDS)
    add_dependencies(${target} ${ARG_DEPENDS})
  endif()
endfunction()

function(cucorr_add_lit_target)
  cmake_parse_arguments(ARG "" "" "SUITES" ${ARGN})

  foreach(suite IN LISTS ARG_SUITES)
    if("${suite}" STREQUAL "all")
      set(SUITE_PATH ${CMAKE_CURRENT_SOURCE_DIR})
      set(TARGET_NAME check-cucorr)
    else()
      set(SUITE_PATH ${CMAKE_CURRENT_SOURCE_DIR}/${suite})
      set(TARGET_NAME check-cucorr-${suite})
    endif()

    if(NOT EXISTS ${SUITE_PATH})
      message(WARNING "Could not find suitable lit test target at ${SUITE_PATH}")
      continue()
    endif()

    cucorr_add_lit_testsuite(${TARGET_NAME}
      "Running the lit suite cucorr::${suite}"
      ${SUITE_PATH}
      ARGS -v -j 1
      PARAMS cucorr_site_config=${CMAKE_CURRENT_BINARY_DIR}/lit.site.cfg
      DEPENDS ${CUCORR_TEST_DEPENDS}
    )
  endforeach()
endfunction()
